#include<iostream>
using namespace std;

int k=0;

void Counting_Sort(int A[],int B[],int n)
{
	int C[k+1];

	for(int i=0;i<=k;i++)
	{
		C[i]=0;
	}
	for(int j=1;j<=n;j++)
	{
		C[A[j]]++;
	}
	for(int i=1;i<=k;i++)
	{
		C[i]+=C[i-1];
	}
	for(int j=n;j>=1;j--)
	{
		B[C[A[j]]]=A[j];

		C[A[j]]=C[A[j]]-1;
	}
}
int main()
{
	int n;
	cout<<"Enter the size of the array :";
	cin>>n;

	int A[n],B[n];

	for(int i=1;i<=n;i++)
	{
		cin>>A[i];
		if(A[i]>k)
		{
			/*It will modify k if an element
			occurs whose value is greater than k*/
			k=A[i];
		}
	}
	Counting_Sort(A,B,n);
	/*It will print the sorted sequence on the
	console*/
	for(int i=1;i<=n;i++)
	{
		cout<<B[i]<<" ";
	}

	cout<<endl;
	return 0;
}
